177 inline GPDSPError GPDSPOutputtableNode::getValueO(int index, GPDSPFloat* value) const noexcept
178 {
179  GPDSPError error(GPDSPERROR_OK);
180 
181  if (0 <= index && index < static_cast<int>(_terminal.size())) {
182  if (_terminal[index].valid) {
183  if (value != NULL) {
184  *value = _terminal[index].value;
185  }
186  }
187  else {
188  error = GPDSPERROR_WAIT;
189  }
190  }
191  else {
192  error = GPDSPERROR_INVALID_RANGE;
193  }
194  return error;
195 }

204  }
205  else {
206  error = GPDSPERROR_INVALID_RANGE;
