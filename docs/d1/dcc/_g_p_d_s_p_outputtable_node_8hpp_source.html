109 inline GPDSPError GPDSPOutputtableNode::getValueO(int index, GPDSPFloat* value) const
110 {
111  GPDSPError error(GPDSPERROR_OK);
112 
113  if (0 <= index && index < static_cast<int>(_terminal.size())) {
114  if (_terminal[index].valid) {
115  if (value != NULL) {
116  *value = _terminal[index].value;
117  }
118  }
119  else {
120  error = GPDSPERROR_WAIT;
121  }
122  }
123  else {
124  error = GPDSPERROR_INVALID_RANGE;
125  }
126  return error;
127 }
